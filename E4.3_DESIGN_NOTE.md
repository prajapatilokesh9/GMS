```prisma
// ============================================================
// EQUIPMENT — Maintenance
// ============================================================
model MaintenanceJob {
  id                        String    @id @default(uuid()) @db.Uuid
  tenantId                  String    @map("tenant_id") @db.Uuid
  inventoryId               String    @map("inventory_id") @db.Uuid
  scheduledDate             DateTime  @map("scheduled_date") @db.Date
  startedAt                 DateTime? @map("started_at")
  completedAt               DateTime? @map("completed_at")
  type                      String    @default("preventive") @db.VarChar(20)       // preventive | corrective | amc
  status                    String    @default("scheduled") @db.VarChar(50)

  // Technician assignment (loose UUID — no FK; technicians may be external)

  assignedToType            String?   @default("internal") @map("assigned_to_type") @db.VarChar(20)  // internal | marketplace (E6)
  assignedToExternalId      String?   @map("assigned_to_external_id") @db.VarChar(100)               // E6 marketplace provider ID
  assignedToExternalName    String?   @map("assigned_to_external_name") @db.VarChar(200)             // E6 marketplace display name

  // Cost tracking
  estimatedCost             Decimal?  @map("estimated_cost") @db.Decimal(10, 2)
  laborCost                 Decimal?  @map("labor_cost") @db.Decimal(10, 2)
  partsCost                 Decimal?  @map("parts_cost") @db.Decimal(10, 2)
  totalCost                 Decimal?  @map("total_cost") @db.Decimal(10, 2)
  currency                  String    @default("INR") @db.VarChar(3)
  invoiceReference          String?   @map("invoice_reference") @db.VarChar(200)
  partsUsed                 Json?     @map("parts_used")                           // [{ partName, qty, unitCost, total }]

  // Approval
  approvalRequired          Boolean   @default(false) @map("approval_required")
  approvedBy                String?   @map("approved_by") @db.Uuid
  approvedAt                DateTime? @map("approved_at")

  // AMC integration (reserved for E4.4 — nullable)
  amcContractId             String?   @map("amc_contract_id") @db.Uuid            // FK to AmcContract (E4.4)
  amcCovered                Boolean?  @map("amc_covered")                         // Whether this job is AMC-covered
  amcClaimReference         String?   @map("amc_claim_reference") @db.VarChar(200) // AMC claim/approval ref
  amcBillingType            String?   @map("amc_billing_type") @db.VarChar(20)   // included | partial | excess
  amcApprovedAmount         Decimal?  @map("amc_approved_amount") @db.Decimal(10, 2) // Pre-approved AMC amount

  // Notes
  description               String?   @db.Text
  outcome                   String?   @db.Text

  // Soft delete
  isActive                  Boolean   @default(true) @map("is_active")
  createdAt                 DateTime  @default(now()) @map("created_at")
  updatedAt                 DateTime  @updatedAt @map("updated_at")
  deletedAt                 DateTime? @map("deleted_at")

  inventory                 EquipmentInventory @relation(fields: [inventoryId], references: [id], onDelete: Restrict)
  tenant                    Tenant             @relation(fields: [tenantId], references: [id], onDelete: Cascade)

  @@index([tenantId, status])
  @@index([tenantId, inventoryId])
  @@index([tenantId, scheduledDate])
  @@index([tenantId, assignedTo])
  @@index([tenantId, amcContractId])
  @@map("maintenance_job")
  @@schema("equipment")
}

model MaintenanceStatusLog {
  id          String   @id @default(uuid()) @db.Uuid
  tenantId    String   @map("tenant_id") @db.Uuid
  jobId       String   @map("job_id") @db.Uuid
  fromStatus  String?  @map("from_status") @db.VarChar(50)
  toStatus    String   @map("to_status") @db.VarChar(50)
  changedBy   String?  @map("changed_by") @db.Uuid
  reason      String?  @map("reason") @db.VarChar(500)
  createdAt   DateTime @default(now()) @map("created_at")

  job     MaintenanceJob @relation(fields: [jobId], references: [id], onDelete: Cascade)
  tenant  Tenant         @relation(fields: [tenantId], references: [id], onDelete: Cascade)

  @@index([jobId])
  @@index([tenantId])
  @@index([tenantId, createdAt])
  @@map("maintenance_status_log")
  @@schema("equipment")
```

## 2. Maintenance Lifecycle State Machine

### States

| State | Description |
|-------|-------------|
| `scheduled` | Job created, not yet started |
| `in_progress` | Technician has begun work |
| `completed` | Work finished successfully (terminal) |
| `cancelled` | Job cancelled before or during work (terminal) |
| `failed` | Work could not be completed — may be retried via `failed → scheduled` |

### Transitions

| # | From | To | Trigger | Allowed By | Reason Required | Guard Condition |
|---|------|----|---------|------------|-----------------|-----------------|
| 1 | scheduled | in_progress | Work begins | maintenance:manage | No | Inventory must be `active`; if `estimatedCost > AUTO_APPROVAL_LIMIT`, must be approved |
| 2 | scheduled | cancelled | Cancelled before start | maintenance:manage | Yes | — |
| 3 | in_progress | completed | Work done successfully | maintenance:manage | No | `totalCost` must be recorded; if cost variance > 20% from estimated, additional approval required |
| 4 | in_progress | cancelled | Cancelled mid-work | maintenance:manage | Yes | — |
| 5 | in_progress | failed | Work could not complete | maintenance:manage | Yes | — |
| 6 | failed | scheduled | Retry with new plan/technician | maintenance:manage | Yes | Inventory must not be `decommissioned` or `lost_stolen` |

**Terminal states:** `completed`, `cancelled` — no outgoing transitions.  
`failed` is **semi-terminal**: can be retried via `failed → scheduled`.

### Inventory Integration

| Maintenance Transition | Inventory Side Effect |
|-----------------------|----------------------|
| `scheduled` → `in_progress` | Auto: `active` → `maintenance` via inventory PATCH (preserves guards + events) |
| `in_progress` → `completed` | Auto: `maintenance` → `active` via inventory PATCH; update `lastMaintenanceAt = completedAt`; recalculate `nextMaintenanceAt` |
| `in_progress` → `cancelled` | Auto: `maintenance` → `active` via inventory PATCH |
| `in_progress` → `failed` | No auto inventory change — equipment stays `maintenance` for retry |
| `failed` → `scheduled` | No inventory change — retry is a new attempt under same issue |
| `scheduled` → `cancelled` | No inventory change |

**`nextMaintenanceAt` formula (same as E4.2):**
```
nextMaintenanceAt = completedAt + INTERVAL 'Y months'
where Y = inventory.maintenanceIntervalMonths ?? 6
```

## 3. Technician Assignment Model

### Fields
- `assignedTo` (UUID, loose reference) — ID of the user/technician responsible  
- `assignedTechnicianName` (Varchar, denormalised) — display name for convenience  
- `assignedToType` (Varchar) — discriminator for the assignment source:  
  - `internal` — assigned to a system user (`assignedTo` = user UUID)  
  - `marketplace` — assigned via external marketplace provider (E6)  

### E6 Marketplace Compatibility

| Field | Purpose |
|-------|---------|
| `assignedToType` | Discriminator: `internal` or `marketplace` |
| `assignedTo` | Internal UUID for `internal` type; marketplace user UUID for `marketplace` type |
| `assignedToExternalId` | Unique provider ID from the marketplace (E6), e.g. `"MP-1042"` |
| `assignedToExternalName` | Display name of the external provider/technician |

The model supports both paths without breaking changes:
- `internal` assignments work today with just `assignedTo`
- `marketplace` assignments become active when E6 introduces the marketplace provider registry
- No FK constraint on `assignedTo` — intentionally loose to accommodate external references

### Rules
- A job **must** have `assignedTo` before transitioning `scheduled → in_progress`
- Reassignment after `in_progress` is allowed only with a reason (audited)

## 4. Cost Tracking & Financial Integrity Rules

### Fields
- `estimatedCost` — pre-maintenance estimate
- `laborCost` — labour charges (set on completion)
- `partsCost` — parts/materials (set on completion)
- `totalCost` — **system-computed** = `laborCost + partsCost`
- `partsUsed` — JSON array `[{ partName, qty, unitCost, total }]`
- `currency` — defaults to `INR`
- `invoiceReference` — optional link to external invoice

### Integrity Rules

| Rule | Behaviour |
|------|-----------|
| **Auto-compute total** | `totalCost` is not user-input; computed as `laborCost + partsCost` before save |
| **Approval threshold** | If `estimatedCost > AUTO_APPROVAL_LIMIT`, `approvalRequired = true`; job cannot start without `approvedBy` + `approvedAt`. `AUTO_APPROVAL_LIMIT` is a **code constant** (`DEFAULT = ₹10,000`) with a documented path to tenant-level config via `tenant.settings.autoApprovalLimit` when a tenant settings system is introduced |
| **Cost overrun** | If `totalCost > estimatedCost × 1.20`, approval required before completion |
| **Completion guard** | `totalCost` must be non-null to transition `in_progress → completed` |
| **Post-completion edits** | Editing `laborCost`/`partsCost` after `completed` requires a reason and is audited |
| **AMC-linked jobs** | If `amcContractId` is set, cost fields may be auto-populated or constrained by contract terms (E4.4) |

## 5. Event Publication Matrix

| Event | Trigger | Payload Key Fields |
|-------|---------|-------------------|
| `maintenance.job.scheduled` | Job created | `{ jobId, inventoryId, serialNumber, catalogueItemId, gymId, type, scheduledDate, assignedTo, estimatedCost }` |
| `maintenance.job.started` | `scheduled → in_progress` | `{ jobId, inventoryId, serialNumber, startedAt }` + emits `equipment.inventory.status.changed` (active→maintenance) |
| `maintenance.job.completed` | `in_progress → completed` | `{ jobId, inventoryId, serialNumber, completedAt, totalCost, laborCost, partsCost, outcome, nextMaintenanceAt }` + emits `equipment.inventory.status.changed` (maintenance→active) + `equipment.inventory.updated` (dates) |
| `maintenance.job.cancelled` | Any → `cancelled` | `{ jobId, inventoryId, serialNumber, reason, previousStatus }` + emits `equipment.inventory.status.changed` if maintenance→active |
| `maintenance.job.failed` | `in_progress → failed` | `{ jobId, inventoryId, serialNumber, reason, outcome }` |

All events published to the **`billing`** exchange (same pattern as E4.2).

## 6. Audit Requirements

### MaintenanceStatusLog (per-job status history)
Every state transition records: `jobId`, `tenantId`, `fromStatus`, `toStatus`, `changedBy`, `reason`, `createdAt`.

### Central Audit Log (audit_service)

| Action | When | Details |
|--------|------|---------|
| `CREATE` | Job created | entityType: `maintenance_job`, entityId, description with type/serial |
| `STATUS_CHANGE` | Every state transition | from→to status, changedBy |
| `UPDATE` | Non-status edits (cost, assignment, notes) | changedFields |
| `UPDATE_COST` | Cost edits after `completed` | old vs new cost values (requires reason) |
| `DELETE` | Soft-delete | — |
| `RESTORE` | Restore after soft-delete | — |

## 7. Integration Points

### With E4.2 Equipment Inventory

| Integration | Detail |
|-------------|--------|
| `inventoryId` FK | `MaintenanceJob.inventoryId → EquipmentInventory.id` (Restrict on delete — cannot delete equipment with active jobs) |
| Status sync | Maintenance transitions auto-PATCH inventory status (see §2 table) |
| Date sync | `completed` → updates `lastMaintenanceAt`, recalculates `nextMaintenanceAt` |
| **Guard: no duplicate active jobs** | Cannot transition `scheduled → in_progress` if another non-terminal job exists for same `inventoryId` |
| **Guard: valid inventory status** | Can only create/schedule jobs for `active` inventory |
| **Guard: terminal inventory** | Cannot create jobs for `decommissioned`, `lost_stolen`, or soft-deleted inventory |

### With E4.4 AMC Contracts (Forward Reference)

| Integration | Detail |
|-------------|--------|
| `amcContractId` FK | `MaintenanceJob.amcContractId → AmcContract.id` (future, nullable) |
| `amcCovered` | Boolean — reserved for E4.4 to mark AMC-covered jobs |
| `amcClaimReference` | Varchar — reserved for E4.4 claim/approval reference number |
| `amcBillingType` | Varchar — reserved for E4.4 billing discriminator: `included`, `partial`, `excess` |
| `amcApprovedAmount` | Decimal — reserved for E4.4 pre-approved amount from AMC provider |
| AMC auto-scheduling | E4.4 may auto-create preventive jobs based on contract terms |
| Cost coverage | E4.4 may constrain cost fields (e.g., covered amount, billing rules) |
| No coupling yet | E4.3 does not depend on E4.4; fields exist for future linking |

## 8. API Surface (draft)

| Method | Path | Description |
|--------|------|-------------|
